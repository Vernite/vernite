export const FullscreenIframe = ({ src }) => {
  setTimeout(() => {
    console.log(document.querySelectorAll('.sbdocs'));
    document.querySelectorAll('.sbdocs').forEach((elem) => {
      elem.style.margin = 0;
      elem.style.padding = 0;
      elem.style.maxWidth = '100%';
      elem.style.backgroundColor = 'white';
    });
  }, 200);
  src = src.replace('{environment}', process.env.STORYBOOK_ENVIRONMENT);
  return <iframe src={src} style={{ width: '100%', height: '100%' }} />;
};
